     21         tokens: &[TokenId],
     22         num_shards: usize,
     23         node_ids: &[String],
     24     ) -> Vec<ContextShard> {
     25         let tokens_per_shard = tokens.len() / num_shards;
     26         
     27         (0..num_shards).map(|i| {
     28             let start = i * tokens_per_shard;
     29             let end = if i == num_shards - 1 {
     30                 tokens.len()
     31             } else {
     32                 (i + 1) * tokens_per_shard
     33             };
     34             
     35             ContextShard {
     36                 shard_id: i as u64,
     37                 node_id: node_ids[i % node_ids.len()].clone(),
     38                 token_start: start,
     39                 token_end: end,
     40                 kv_data: vec![],  // 后续填充
     41                 shard_hash: compute_hash(&tokens[start..end]),
     42             }
     43         }).collect()
     44     }

      12 #[derive(Debug, Clone)]
      13 pub enum CompressionAlgorithm {
      14     /// 量化（FP16 -> INT8）
      15     Quantization,
      16     /// PCA 降维
      17     Pca { components: usize },
      18     /// 稀疏化（保留 top-k）
      19     Sparsification { top_k: usize },
      20 }
      21 
      22 impl KvCompressor {
      23     pub fn new(algorithm: CompressionAlgorithm, target_ratio: f32) -> Self {
      24         KvCompressor {
      25             algorithm,
      26             target_ratio,
      27         }
      28     }
      29     
      30     /// 压缩 KV Cache
      31     pub fn compress(&self, kv: Array2<f32>) -> Result<CompressedKv, String> {
      32         match &self.algorithm {
      33             CompressionAlgorithm::Quantization => {
      34                 self.quantize(kv)
      35             }
      36             CompressionAlgorithm::Pca { components } => {
      37                 self.pca_compress(kv, *components)
      38             }
      39             CompressionAlgorithm::Sparsification { top_k } => {
      40                 self.sparse_compress(kv, *top_k)
      41             }
      42         }
      43     }
      44     
      45     /// 解压缩
      46     pub fn decompress(&self, compressed: CompressedKv) -> Result<Array2<f32>, String> {
      47         match compressed {
      48             CompressedKv::Quantized(q) => self.dequantize(q),
      49             CompressedKv::Pca { coefficients, components } => {
      50                 self.pca_decompress(coefficients, components)
      51             }
      52             CompressedKv::Sparse { indices, values, shape } => {
      53                 self.sparse_decompress(indices, values, shape)
      54             }
      55         }
      56     }
      57     
      58     fn quantize(&self, kv: Array2<f32>) -> Result<CompressedKv, String> {
      59         // FP32 -> INT8 量化
      60         let min_val = kv.iter().cloned().fold(f32::INFINITY, f32::min);
      61         let max_val = kv.iter().cloned().fold(f32::NEG_INFINITY, f32::max);
      62         
      63         let scale = (max_val - min_val) / 255.0;
      64         let quantized: Vec<u8> = kv.iter()
      65             .map(|&v| ((v - min_val) / scale) as u8)
      66             .collect();
      67         
      68         Ok(CompressedKv::Quantized(QuantizedKv {
      69             data: quantized,
      70             min: min_val,
      71             max: max_val,
      72             shape: kv.dim(),
      73         }))
      74     }
      75     
      76     fn dequantize(&self, q: QuantizedKv) -> Result<Array2<f32>, String> {
      77         let scale = (q.max - q.min) / 255.0;
      78         let dequantized: Vec<f32> = q.data.iter()
      79             .map(|&v| q.min + v as f32 * scale)
      80             .collect();
      81         
      82         Ok(Array2::from_shape_vec(q.shape, dequantized)?)
      83     }
      84     
      85     fn pca_compress(&self, kv: Array2<f32>, components: usize) -> Result<CompressedKv, String> {
      86         // 简化版 PCA（实际应该用 linfa 或 smartcore）
      87         // 这里只做投影，不训练
      88         Err("PCA training not implemented".to_string())
      89     }
      90     
      91     fn sparse_compress(&self, kv: Array2<f32>, top_k: usize) -> Result<CompressedKv, String> {
      92         // 保留绝对值最大的 top_k 个元素
      93         let mut indexed: Vec<_> = kv.iter()
      94             .enumerate()
      95             .map(|(i, &v)| (i, v))
      96             .collect();
      97         
      98         indexed.sort_by(|a, b| {
      99             b.1.abs().partial_cmp(&a.1.abs()).unwrap()
     100         });
     101         
     102         let selected: Vec<_> = indexed.into_iter().take(top_k).collect();
     103         
     104         let indices: Vec<usize> = selected.iter().map(|(i, _)| *i).collect();
     105         let values: Vec<f32> = selected.iter().map(|(_, v)| *v).collect();
     106         
     107         Ok(CompressedKv::Sparse {
     108             indices,
     109             values,
     110             shape: kv.dim(),
     111         })
     112     }
     113     
     114     fn sparse_decompress(&self, indices: Vec<usize>, values: Vec<f32>, shape: (usize, usize)) -> Result<Array2<f32>, String> {
     115         let mut kv = Array2::zeros(shape);
     116         
     117         for (&idx, &val) in indices.iter().zip(values.iter()) {
     118             kv.as_slice_mut().unwrap()[idx] = val;
     119         }
     120         
     121         Ok(kv)
     122     }

       4 /// PBFT 共识状态
       5 #[derive(Debug, Clone, PartialEq)]
       6 pub enum PbftPhase {
       7     PrePrepare,
       8     Prepare,
       9     Commit,
      10 }
      11 
      12 /// PBFT 消息
      13 #[derive(Debug, Clone)]
      14 pub struct PbftMessage {
      15     pub view: u64,
      16     pub sequence: u64,
      17     pub phase: PbftPhase,
      18     pub node_id: String,
      19     pub digest: String,  // 请求哈希
      20     pub signature: String,
      21 }
      22 
      23 /// PBFT 共识引擎
      24 pub struct PbftEngine {
      25     node_id: String,
      26     replicas: Vec<String>,  // 所有副本节点
      27     f: usize,  // 容错节点数（总节点数 n = 3f + 1）
      28     view: u64,
      29     sequence: u64,
      30     log: HashMap<u64, PbftLogEntry>,
      31 }
      32 
      33 #[derive(Debug)]
      34 struct PbftLogEntry {
      35     digest: String,
      36     pre_prepare_received: bool,
      37     prepare_count: HashMap<String, usize>,  // node_id -> count
      38     commit_count: HashMap<String, usize>,
      39     committed: bool,
      40 }
      41 
      42 impl PbftEngine {
      43     pub fn new(node_id: String, replicas: Vec<String>) -> Self {
      44         let n = replicas.len();
      45         let f = (n - 1) / 3;  // n = 3f + 1
      46         
      47         PbftEngine {
      48             node_id,
      49             replicas,
      50             f,
      51             view: 0,
      52             sequence: 0,
      53             log: HashMap::new(),
      54         }
      55     }
      56     
      57     /// 处理预准备消息（主节点发送）
      58     pub fn handle_pre_prepare(
      59         &mut self,
      60         digest: &str,
      61         request: &str,
      62     ) -> Option<PbftMessage> {
      63         // 验证：是主节点且 view 正确
      64         if !self.is_primary() {
      65             return None;
      66         }
      67         
      68         self.sequence += 1;
      69         
      70         self.log.insert(self.sequence, PbftLogEntry {
      71             digest: digest.to_string(),
      72             pre_prepare_received: true,
      73             prepare_count: HashMap::new(),
      74             commit_count: HashMap::new(),
      75             committed: false,
      76         });
      77         
      78         // 广播 Prepare 消息
      79         Some(PbftMessage {
      80             view: self.view,
      81             sequence: self.sequence,
      82             phase: PbftPhase::Prepare,
      83             node_id: self.node_id.clone(),
      84             digest: digest.to_string(),
      85             signature: self.sign(digest),
      86         })
      87     }
      88     
      89     /// 处理 Prepare 消息
      90     pub fn handle_prepare(
      91         &mut self,
      92         message: PbftMessage,
      93     ) -> Option<PbftMessage> {
      94         // 验证签名和 view
      95         if !self.verify_signature(&message) {
      96             return None;
      97         }
      98         
      99         if message.view != self.view {
     100             return None;
     101         }
     102         
     103         // 记录 Prepare
     104         let entry = self.log.entry(message.sequence).or_insert_with(|| PbftLogEntry {
     105             digest: message.digest.clone(),
     106             pre_prepare_received: false,
     107             prepare_count: HashMap::new(),
     108             commit_count: HashMap::new(),
     109             committed: false,
     110         });
     111         
     112         *entry.prepare_count.entry(message.node_id).or_insert(0) += 1;
     113         
     114         // 检查是否收到 2f+1 个 Prepare（包括自己）
     115         let prepare_total: usize = entry.prepare_count.values().sum();
     116         if prepare_total >= 2 * self.f + 1 && !entry.committed {
     117             // 进入 Commit 阶段
     118             entry.committed = true;
     119             
     120             Some(PbftMessage {
     121                 view: self.view,
     122                 sequence: message.sequence,
     123                 phase: PbftPhase::Commit,
     124                 node_id: self.node_id.clone(),
     125                 digest: message.digest,
     126                 signature: self.sign(&message.digest),
     127             })
     128         } else {
     129             None
     130         }
     131     }
     132     
     133     /// 处理 Commit 消息
     134     pub fn handle_commit(
     135         &mut self,
     136         message: PbftMessage,
     137     ) -> bool {
     138         // 验证
     139         if !self.verify_signature(&message) {
     140             return false;
     141         }
     142         
     143         let entry = match self.log.get_mut(&message.sequence) {
     144             Some(e) => e,
     145             None => return false,
     146         };
     147         
     148         *entry.commit_count.entry(message.node_id).or_insert(0) += 1;
     149         
     150         // 检查是否收到 2f+1 个 Commit
     151         let commit_total: usize = entry.commit_count.values().sum();
     152         if commit_total >= 2 * self.f + 1 {
     153             // 共识达成！
     154             true
     155         } else {
     156             false
     157         }
     158     }
     159     
     160     fn is_primary(&self) -> bool {
     161         self.replicas[self.view as usize % self.replicas.len()] == self.node_id
     162     }
     163     
     164     fn sign(&self, digest: &str) -> String {
     165         // 实际应该用 Ed25519 签名
     166         format!("sig_{}_{}", self.node_id, digest)
     167     }
     168     
     169     fn verify_signature(&self, message: &PbftMessage) -> bool {
     170         // 实际应该验证 Ed25519 签名
     171         message.signature.starts_with(&format!("sig_{}", message.node_id))
     172     }
